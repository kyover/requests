// 简称: TB_COMPANYINFO_NUMBEROFSHAREHOLDERS
// 名称: 股东人数
// 类别: 公式应用
// 类型: 内建应用
//------------------------------------------------------------------------
Params
	Enum<String>    BackType(["基础数据回溯","BAR回溯"]);//回溯类型
	Numeric         BackLength(0);//回溯周期
Vars
	Dic<Array<Numeric>>		MyVar("TB_COMPANYINFO_NUMBEROFSHAREHOLDERS");
Events
    OnInit()
    {
        Range[0:DataCount()-1]
        {
            If(BackType == "基础数据回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackTime);
            Else If(BackType == "BAR回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackBar);
        }
    }

    OnBar(ArrayRef<Integer> indexs)
    {
        Range[0:DataSourceSize()-1]
        {
			PlotNumeric("统计日期",MyVar[BackLength][0]);
			PlotNumeric("总户数",MyVar[BackLength][1]);
			PlotNumeric("A股户数",MyVar[BackLength][2]);
			PlotNumeric("户均持股数",MyVar[BackLength][3]);
			PlotNumeric("A股户均持股数",MyVar[BackLength][4]);
			PlotNumeric("前十大股东持股比例",MyVar[BackLength][5]);
		}
	}
