// 简称: TB_COMPANYINFO_BUSINESSANALYSISINDUSTRY
// 名称: 经营分析行业分类
// 类别: 公式应用
// 类型: 内建应用
//------------------------------------------------------------------------
Params
	Enum<String>    BackType(["基础数据回溯","BAR回溯"]);//回溯类型
	Numeric         BackLength(0);//回溯周期
	Numeric         index(0);//选择数据序号
Vars
	Dic<Array<Array<String>>>		MyVar("TB_COMPANYINFO_BUSINESSANALYSISINDUSTRY");
Events
    OnInit()
    {
        Range[0:DataCount()-1]
        {
            If(BackType == "基础数据回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackTime);
            Else If(BackType == "BAR回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackBar);
        }
    }

    OnBar(ArrayRef<Integer> indexs)
    {
        Range[0:DataSourceSize()-1]
        {
			PlotString("主营构成",MyVar[BackLength][index][0]);
			Plotnumeric("主营业务收入",Value(MyVar[BackLength][index][1]));
			Plotnumeric("主营业务成本",Value(MyVar[BackLength][index][2]));
			Plotnumeric("上年同期主营业务收入(元)",Value(MyVar[BackLength][index][3]));
			Plotnumeric("上年同期主营业务成本(元)",Value(MyVar[BackLength][index][4]));
			Plotnumeric("主营业务收入同比",Value(MyVar[BackLength][index][5]));
			Plotnumeric("主营业务成本同比",Value(MyVar[BackLength][index][6]));
		}
	}
