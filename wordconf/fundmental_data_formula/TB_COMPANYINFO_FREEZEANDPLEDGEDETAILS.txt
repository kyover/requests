// 简称: TB_COMPANYINFO_FREEZEANDPLEDGEDETAILS
// 名称: 股东股权冻结和质押明细
// 类别: 公式应用
// 类型: 内建应用
//------------------------------------------------------------------------
Params
	Enum<String>    BackType(["基础数据回溯","BAR回溯"]);//回溯类型
	Numeric         BackLength(0);//回溯周期
Vars
	Dic<Array<Numeric>>		MyVar("TB_COMPANYINFO_FREEZEANDPLEDGEDETAILS");
Events
    OnInit()
    {
        Range[0:DataCount()-1]
        {
            If(BackType == "基础数据回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackTime);
            Else If(BackType == "BAR回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackBar);
        }
    }

    OnBar(ArrayRef<Integer> indexs)
    {
        Range[0:DataSourceSize()-1]
        {
			PlotNumeric("截止日期",MyVar[BackLength][0]);
			PlotNumeric("待回购余量(万股)",MyVar[BackLength][1]);
			PlotNumeric("待回购无限售条件余量(万股)",MyVar[BackLength][2]);
			PlotNumeric("待回购有限售条件余量(万股)",MyVar[BackLength][3]);
			PlotNumeric("质押笔数",MyVar[BackLength][4]);
			PlotNumeric("质押比例",MyVar[BackLength][5]);
		}
	}
