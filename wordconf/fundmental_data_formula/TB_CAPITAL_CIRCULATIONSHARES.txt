//------------------------------------------------------------------------
// 简称: TB_CAPITAL_CIRCULATIONSHARES
// 名称: 流通股份
// 类别: 公式应用
// 类型: 内建应用
//------------------------------------------------------------------------
Params
	Enum<String>    BackType(["基础数据回溯","BAR回溯"]);//回溯类型
	Numeric         BackLength(0);//回溯周期
Vars
	Dic<Array<Numeric>>		MyVar("TB_CAPITAL_CIRCULATIONSHARES");
Events
    OnInit()
    {
        Range[0:DataCount()-1]
        {
            If(BackType == "基础数据回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackTime);
            Else If(BackType == "BAR回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackBar);
        }
    }

    OnBar(ArrayRef<Integer> indexs)
    {
        Range[0:DataSourceSize()-1]
        {
			PlotNumeric("报告时间",MyVar[BackLength][0]);
			PlotNumeric("流通股份(股)",MyVar[BackLength][1]);
			PlotNumeric("已上市流通A股(包含高管股)(股)",MyVar[BackLength][2]);
			PlotNumeric("战略投资者配售持股(股)",MyVar[BackLength][3]);
			PlotNumeric("一般法人配售持股(股)",MyVar[BackLength][4]);
			PlotNumeric("基金配售持股(股)",MyVar[BackLength][5]);
			PlotNumeric("增发未上市(股)",MyVar[BackLength][6]);
			PlotNumeric("配股未上市(股)",MyVar[BackLength][7]);
			PlotNumeric("其他流通股份(股)",MyVar[BackLength][8]);
			PlotNumeric("有限售流通A股(股)",MyVar[BackLength][9]);
			PlotNumeric("有限售流通股中职工股(股)",MyVar[BackLength][10]);
			PlotNumeric("B股_旧(股)",MyVar[BackLength][11]);
			PlotNumeric("未流通B股_旧",MyVar[BackLength][12]);
			PlotNumeric("有限售B股(股)",MyVar[BackLength][13]);
			PlotNumeric("外资持A股(股)",MyVar[BackLength][14]);
		}
	}
