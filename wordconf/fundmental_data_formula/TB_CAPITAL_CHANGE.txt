//------------------------------------------------------------------------
// 简称: TB_CAPITAL_CHANGE
// 名称: 股本变动
// 类别: 公式应用
// 类型: 内建应用
//------------------------------------------------------------------------
Params
	Enum<String>    BackType(["基础数据回溯","BAR回溯"]);//回溯类型
	Numeric         BackLength(0);//回溯周期
Vars
	Dic<Array<String>>		MyVar("TB_CAPITAL_CHANGE");
Events
    OnInit()
    {
        Range[0:DataCount()-1]
        {
            If(BackType == "基础数据回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackTime);
            Else If(BackType == "BAR回溯")
                SetDicFlag(MyVar, Enum_DicFlag_BackBar);
        }
    }

    OnBar(ArrayRef<Integer> indexs)
    {
        Range[0:DataSourceSize()-1]
        {
			Plotnumeric("报告时间",Value(MyVar[BackLength][0]));
			PlotString("股本变动原因类别",MyVar[BackLength][1]);
			PlotString("股本变动原因说明",MyVar[BackLength][2]);
		}
	}
